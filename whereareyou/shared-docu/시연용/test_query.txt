DROP TABLE EMPLOYEE
DROP TABLE PROJECT
CREATE MEMORY TABLE PUBLIC.PROJECT(PRJ_SEQ VARCHAR(50) NOT NULL PRIMARY KEY,PRJ_NM VARCHAR(100) NOT NULL,PRJ_ADDR VARCHAR(150) NOT NULL,PRJ_REP_ID_EMP_ID VARCHAR(10) NOT NULL,PRJ_POS_X INTEGER NOT NULL,PRJ_POS_Y INTEGER NOT NULL)
CREATE MEMORY TABLE PUBLIC.EMPLOYEE(EMP_ID VARCHAR(10) NOT NULL PRIMARY KEY,EMP_NM VARCHAR(30) NOT NULL,EMP_ENT_SEQ VARCHAR(10),EMP_DEPT_NM VARCHAR(100) NOT NULL,EMP_SEND_CODE VARCHAR(2) NOT NULL,EMP_PRJ_SEQ VARCHAR(50),CONSTRAINT FK_EMPLOYEE FOREIGN KEY(EMP_PRJ_SEQ) REFERENCES PUBLIC.PROJECT(PRJ_SEQ) ON DELETE SET NULL)
